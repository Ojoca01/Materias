--1 Liste o login dos usuários cadastrados a partir de 1990, ordenados pelo login em ordem decrescente

SELECT login FROM usuario 
WHERE data_cadastro >= '1990-01-01'
ORDER BY  login DESC;

--2 Para cada evento cuja empresa é 'TEC', liste: 
o e-mail do cliente que comprou ingresso
o login do organizador responsável
o título do evento

SELECT u.email,og.login,e.titulo 
FROM e c
JOIN evento e
JOIN organizacao o USING(id_evento)
JOIN organizador o USING(id_organizador)
JOIN ingresso i USING(id_evento)
JOIN cliente c USING(id_cliente)
JOIN usuario u ON c.login=u.login
WHERE e.titulo = 'TEC';

--3 Liste o título do evento e o título do seu evento anterior, somente daqueles eventos cujo evento anterior existe e pertence à mesma empresa.

SELECT e.titulo , ea.titulo
FROM evento e
JOIN evento ae
ON  (e.evento = evento_anterior)
WHERE e.empresa=ae.empresa;

-- 4 Liste o login dos usuários que são clientes mas nunca compraram ingresso.

SELECT c.login
FROM usuario u
JOIN cliente c USING (login) 
LEFT JOIN ingresso i
USING (id_cliente) 
WHERE i.id_cliente IS NULL;

-- 5 Liste o título do evento, a empresa responsável e o valor médio dos ingressos de cada evento.

SELECT e.titulo,e.empresa, AVG(i.valor_ingresso)
FROM evento e
JOIN ingresso i USING (id_evento)
GROUP BY titulo,empresa;

-- 6 Liste o id do organizador e sua empresa, considerando apenas organizadores que organizaram eventos com mais de 5 ingressos vendidos.

SELECT og.id_organizador, og.empresa 
FROM organizador og 
JOIN organizacao o USING(id_organizador)
JOIN ingresso i USING (id_evento)
GROUP BY  og.id_organizador, og.empresa 
HAVING COUNT(i.id_cliente) > 5; 

-- 7 Para cada empresa que possui organizadores associados a eventos:

calcule a média salarial dos organizadores da empresa (media_salario)
calcule a média do valor dos ingressos vendidos em seus eventos (media_ingresso)

SELECT ba.id_empresa, AVG(ba.salario) AS media_salario, AVG(bc.valor_ingresso) AS media_ingresso
FROM (
    SELECT o.empresa AS id_empresa,e.salario 
    FROM organizador o/
    JOIN emprego e
    ON (o.emprego =e.id_emprego)
) ba
JOIN
    (
    SELECT o.empresa AS id_empresa,i.valor_ingresso 
    FROM organizador o
    JOIN organizacao org USING (id_organizador)
    JOIN ingresso i USING (id_evento)
) bc
USING (id_empresa)
GROUP BY ba.id_empresa;
